GER_imperial_ambition_up = {

	if = {
		limit = {  has_idea = GER_imperial_territories_some }
		swap_ideas = {
			remove_idea = GER_imperial_territories_some
			add_idea = GER_imperial_territories_reclaimed
		}
	}

	if = {
		limit = { has_idea = GER_imperial_territories_none }
		swap_ideas = {
			remove_idea = GER_imperial_territories_none
			add_idea = GER_imperial_territories_some
		}
	}

	if = {
		limit = { has_idea = GER_imperial_territories_lost }
		swap_ideas = {
			remove_idea = GER_imperial_territories_lost
			add_idea = GER_imperial_territories_none
		}
	}
}

GER_imperial_ambition_down = {

	if = {
		limit = { has_idea = GER_imperial_territories_none }
		swap_ideas = {
			remove_idea = GER_imperial_territories_none
			add_idea = GER_imperial_territories_lost
		}
	}
	if = {
		limit = { has_idea = GER_imperial_territories_some }
		swap_ideas = {
			remove_idea = GER_imperial_territories_some
			add_idea = GER_imperial_territories_none
		}
	}
	if = {
		limit = { has_idea = GER_imperial_territories_some }
		swap_ideas = {
			remove_idea = GER_imperial_territories_some
			add_idea = GER_imperial_territories_reclaimed
		}
	}
}
