ARG_be_nice_to_germany_early = {

		original_tag = ARG
	}

		#original_tag = ARG
		date < 1937.3.1
		threat < 0.16
		GER = {
			has_war = no
		}
		country_exists = GER
		GER = {
			OR = {
				NOT = {
					has_government = democratic
					has_government = communism
				}
			}
		}
		ARG = {

				has_government = democratic

		}
	}

	abort_when_not_enabled = yes
	

	ai_strategy = {
		type = befriend
		id = "GER"	
		value = 15
	}

}
